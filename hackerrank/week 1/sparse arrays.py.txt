def matchingStrings(strings, queries):
    result = []
    for word in queries:
        result.append(strings.count(word))
    return result


n = int(input())
strings = []
for i in range(n):
    strings.append(input())

q = int(input())
queries = []
for i in range(q):
    queries.append(input())

ans = matchingStrings(strings, queries)

for a in ans:
    print(a)
