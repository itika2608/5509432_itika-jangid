def plusMinus(arr):
    total = len(arr)
    positives = 0
    negatives = 0
    zeros = 0

    for value in arr:
        if value > 0:
            positives += 1
        elif value < 0:
            negatives += 1
        else:
            zeros += 1

    print(f"{positives / total:.6f}")
    print(f"{negatives / total:.6f}")
    print(f"{zeros / total:.6f}")

if __name__ == "__main__":
    n = int(input())
    numbers = list(map(int, input().split()))
    plusMinus(numbers)
